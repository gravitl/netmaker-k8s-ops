1. Get the application URL by running these commands:

{{- if .Values.service.proxy.enabled }}
  Proxy Service:
  export POD_NAME=$(kubectl get pods --namespace {{ include "netmaker-k8s-ops.namespace" . }} -l "app.kubernetes.io/name={{ include "netmaker-k8s-ops.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ include "netmaker-k8s-ops.namespace" . }} port-forward $POD_NAME 8085:8085
{{- end }}

2. Check the operator status:
  kubectl get pods --namespace {{ include "netmaker-k8s-ops.namespace" . }} -l "app.kubernetes.io/name={{ include "netmaker-k8s-ops.name" . }}"

3. View operator logs:
  kubectl logs --namespace {{ include "netmaker-k8s-ops.namespace" . }} -l "app.kubernetes.io/name={{ include "netmaker-k8s-ops.name" . }}" -c manager

{{- if .Values.netclient.enabled }}
4. View netclient sidecar logs:
  kubectl logs --namespace {{ include "netmaker-k8s-ops.namespace" . }} -l "app.kubernetes.io/name={{ include "netmaker-k8s-ops.name" . }}" -c netclient
{{- end }}
